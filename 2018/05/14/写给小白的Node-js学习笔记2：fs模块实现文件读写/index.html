з”Ёnode.jsиҜ»еҶҷж–Үд»¶ж—¶пјҢйңҖиҰҒеј•е…ҘдёҖдёӘж–°зҡ„жЁЎеқ—fs(file systemж–Үд»¶зі»з»ҹпјүгҖӮж–Үд»¶зі»з»ҹзҡ„ж“ҚдҪңйғҪжңүејӮжӯҘе’ҢеҗҢжӯҘзҡ„еҪўејҸпјҢжЎҲдҫӢдҪҝз”ЁејӮжӯҘж–№жі•пјҢйҳІжӯўйҳ»еЎһгҖӮ

 const fs = require('fs');

  //1. иҜ»еҸ–ж–Үд»¶
  fs.readFile('aaa.txt', function(err, data) {
    if(err) {
      console.log('иҜ»еҸ–еӨұиҙҘ');
    } else {
      //ж–Үд»¶дј иҫ“йҮҮз”ЁдәҢиҝӣеҲ¶пјҢеҰӮжһңиҰҒиҫ“еҮәдёәж–Үеӯ—пјҢйңҖиҰҒдҪҝз”ЁtoString()ж–№жі•иҪ¬жҚўдёҖдёӢ
      console.log(data.toString());
    }
  })

  //2. еҶҷе…Ҙж–Үд»¶
  fs.writeFile('bbb.txt', 'hello world', function(err) {
    console.log(err);
  })
  //иҝҷж ·е°ұиғҪж”№еҶҷbbb.txtеҶ…зҡ„еҶ…е®№пјҢз¬¬дәҢдёӘеҸӮж•°дёәдҝ®ж”№зҡ„еҶ…е®№
  //еҰӮжһңжІЎжңүbbb.txtж–Үд»¶пјҢйӮЈд№ҲдјҡеҲӣе»әдёҖдёӘж–°зҡ„bbb.txtж–Үд»¶пјҢеҶ…е®№дёәз¬¬дәҢдёӘеҸӮж•°

ж–Үд»¶иҜ»еҸ–еҸҜд»Ҙй…ҚеҗҲhttpжЁЎеқ—дҪҝз”ЁгҖӮ

const http = require('http');
const fs = require('fs');

http.createServer(function(req, res) {
  //еңЁжөҸи§ҲеҷЁиҜ»еҸ–з«ҜеҸЈеҗҺпјҢеҸҜеңЁз«ҜеҸЈеҗҺж·»еҠ ж–°зҡ„иҜ·жұӮзҡ„ж•°жҚ®ең°еқҖ
  //жңҚеҠЎеҷЁеҸҜд»ҘйҖҡиҝҮreq.urlиҺ·еҸ–жөҸи§ҲеҷЁз«Ҝиҫ“е…Ҙзҡ„ең°еқҖ
  //еҰӮжөҸи§ҲеҷЁз«Ҝдёәlocalhost:8080/1.html, req.urlе°ұдёә1.html
  //fileNameдёәfsжЁЎеқ—иҜ»еҸ–ж•°жҚ®ж—¶зҡ„ең°еқҖпјҢйңҖиҰҒиҖғиҷ‘еңЁжң¬ең°зҡ„зӣёеҜ№и·Ҝеҫ„пјҢдҫӢеӯҗдёӯ1.htmlеңЁfileж–Үд»¶еӨ№дёӢ
  let fileName = 'file/' + req.url;

  fs.readFile(fileName, function(err, data) {
    if(err) {
      res.write('404');
    } else {
      //иҝҷйҮҢжҳҜжңҚеҠЎеҷЁдј йҖ’з»ҷжөҸи§ҲеҷЁзҡ„ж•°жҚ®пјҢдёҚйңҖиҰҒз”ЁtoString()и§ЈжһҗпјҢжөҸи§ҲеҷЁдјҡиҮӘиЎҢи§Јжһҗ
      //еҰӮжһңжҳҜзӣҙжҺҘз”Ёе‘Ҫд»ӨиЎҢиҫ“еҮәпјҢеҲҷйңҖиҰҒи§ЈжһҗпјҢеҗҰеҲҷжҳҜдәҢиҝӣеҲ¶ж•°жҚ®
      res.write(data);
    }
    res.end();
  })
}).listen(8080);

  • йҰ–е…Ҳж·»еҠ дёӨдёӘж–Үд»¶жқҘз»ҷжҲ‘д»¬дёҖдјҡе„ҝиҜ»еҸ–пјҢжҲ‘еңЁж №зӣ®еҪ•дёӢж·»еҠ дәҶдёҖдёӘdataж–Үд»¶еӨ№пјҢйҮҢйқўж”ҫдәҶеҫҲз®ҖеҚ•зҡ„дёӨдёӘж–Үд»¶1.txtе’Ң2.json
const fs = require("fs");

// 1. е®ҡд№үиҜ»еҸ–ж–Үд»¶зҡ„еҮҪж•°пјҢдҪҝд№Ӣиҝ”еӣһдёҖдёӘPromiseеҜ№иұЎ
const readFile = function(fileName) {
  return new Promise((resolve, reject) => {
    fs.readFile(fileName, (err, data) => {
      if(err) return reject(err);
      resolve(data);
    })
  })
}
// 2. е°ҶйңҖиҰҒиҜ»еҸ–зҡ„ж–Үд»¶з”ЁasyncеҮҪж•°зҡ„еҪўејҸеҶҷеҮәжқҘ
const asyncReadFile = async function() {
  const f1 = await readFile('data/1.txt');
  const f2 = await readFile('data/2.json');
  console.log(f1.toString());
  console.log(f2.toString());
}

// 3. asyncеҮҪж•°е’ҢgeneratorеҮҪж•°зҡ„дёҖеӨ§еҢәеҲ«е°ұжҳҜasyncеҮҪж•°еҸҜд»ҘзӣҙжҺҘи°ғз”Ё
asyncReadFile();
